Pengantar Polimorfisme

Dart menyediakan beberapa mekanisme inisialisasi objek untuk berbagai kebutuhan arsitektur:

  1. Arti: "Banyak bentuk" (poly = banyak, morph = bentuk)
  2. Konsep: Objek dari subclass dapat diperlakukan sebagai objek dari superclass
  3. Parameterized Constructor: ClassName(this.param)
  4. Contoh: Semua Kucing , Anjing , Burung adalah Hewan
  5. i.Compile-time polymorphism: Method overloading (tidak didukung Dart)
  6. ii. Runtime polymorphism: Method overriding (didukung Dart)

Perbedaan extends vs implements

Aspect extends implements
Hubungan "is-a" (adalah) "can-do" (bisa melakukan)
Inheritance Mewarisi implementasi ✓ Hanya kontrak ✗
Method Bisa concrete/abstract Semua harus di-implement ✗
Single/Multiple Ya (single inheritance) ✓ Banyak (multiple) ✓
Keyword extends implements
Contoh class MobilListrik extends AlatTransportasi MobilListrik IS-A AlatTransportasi class Elang implements Terbang Elang CAN-DO Terbang

class MathHelper {
  static int factorial(int n) {
    if (n < 0) throw ArgumentError('Tidak bisa menghitung faktorial bilangan negatif');
    if (n == 0 || n == 1) return 1;
    int hasil = 1;
    for (int i = 2; i <= n; i++) {
      hasil *= i;
    }
    return hasil;
  }
  
  static bool isPrima(int n) {
    if (n <= 1) return false;
    if (n == 2) return true;
    if (n % 2 == 0) return false;
    for (int i = 3; i * i <= n; i += 2) {
      if (n % i == 0) return false;
    }
    return true;
  }
  
  static double roundToDecimals(double value, int decimalPlaces) {
    double faktor = 1;
    for (int i = 0; i < decimalPlaces; i++) {
      faktor *= 10;
    }
    return (value * faktor).round() / faktor;
  }
  
  static int randomInt(int min, int max) {
    return min + (DateTime.now().millisecondsSinceEpoch % (max - min + 1));
  }
  
  static double randomDouble(double min, double max) {
    return min + (DateTime.now().millisecondsSinceEpoch % 10000) / 10000 * (max - min);
  }
  
  static double rataRata(List angka) {
    if (angka.isEmpty) return 0;
    double sum = 0;
    for (var a in angka) {
      sum += a;
    }
    return sum / angka.length;
  }
  
  static double median(List angka) {
    if (angka.isEmpty) return 0;
    List sorted = List.from(angka)..sort();
    int tengah = sorted.length ~/ 2;
    if (sorted.length % 2 == 0) {
      return (sorted[tengah - 1] + sorted[tengah]) / 2;
    } else {
      return sorted[tengah];
    }
  }
}

void main() {
  print('========== SOAL 5 ==========');
  
  print('Factorial 6: ${MathHelper.factorial(6)}');
  print('Factorial 8: ${MathHelper.factorial(8)}');
  print('Apakah 19 prima? ${MathHelper.isPrima(19)}');
  print('Apakah 21 prima? ${MathHelper.isPrima(21)}');
  print('2.71828 ke 3 desimal: ${MathHelper.roundToDecimals(2.71828, 3)}');
  print('Random Int (5-15): ${MathHelper.randomInt(5, 15)}');
  print('Random Double (0-1): ${MathHelper.randomDouble(0, 1)}');
  
  List angkaList = [65, 80, 95, 70, 85, 100];
  print('List angka: $angkaList');
  print('Rata-rata: ${MathHelper.rataRata(angkaList)}');
  print('Median: ${MathHelper.median(angkaList)}');
}

  void displayDuration() {
    int menit = durasi ~/ 60;

    print('Durasi: ${menit.toString().padLeft(2, '0')}:${detik.toString().padLeft(2, '0')}');
  }

abstract interface class PlayableContent {
  void play();
  void resume();
}

abstract interface class PausableContent {
  void pause();
  void resumeFromPause();
}

class AudioTrack implements MusicPlayer, PlayableContent, PausableContent {
  @override
  String judul;
  
  @override
  int durasi;
  
  String formatAudioTrack;
  int kbps;
  bool isPlaying = false;
  bool isPaused = false;
  
  AudioTrack({
    required this.judul,
    required this.durasi,
    required this.formatAudioTrack,
    required this.kbps,
  });
  
  @override
  void play() {
    if (isPaused) {
      resume();
    } else {
      isPlaying = true;
      isPaused = false;
      print('🎵 Memutar audio: $judul ($formatAudioTrack, ${kbps}kbps)');
    }
  }
  
  @override
  void resume() {
    if (isPaused) {
      isPlaying = true;
      isPaused = false;
      print('🎵 Melanjutkan audio: $judul');
    }
  }
  
  @override
  void stop() {
    isPlaying = false;
    isPaused = false;
    print('⏹️ Menghentikan audio: $judul');
  }
  
  @override
  void pause() {
    if (isPlaying) {
      isPlaying = false;
      isPaused = true;
      print('⏸️ Menjeda audio: $judul');
    }
  }
  
  @override
  void resumeFromPause() {
    if (isPaused) {
      isPlaying = true;
      isPaused = false;
      print('▶️ Melanjutkan audio dari jeda: $judul');
    }
  }
  
  @override
  void showInfo() {
    print('═══════════════════════════════');
    print('📻 AUDIO TRACK');
    print('═══════════════════════════════');
    print('Judul: $judul');
    print('Format: $formatAudioTrack');
    print('Kualitas: ${kbps}kbps');
    displayDuration();
    print('Status: ${isPlaying ? "Playing" : (isPaused ? "Paused" : "Stopped")}');
    print('═══════════════════════════════');
  }
  
  @override
  void displayDuration() {
    int menit = durasi ~/ 60;
    int detik = durasi % 60;
    print('Durasi: ${menit.toString().padLeft(2, '0')}:${detik.toString().padLeft(2, '0')}');
  }
  
  void setVolume(int volume) {
    print('🔊 Mengatur volume ke $volume%');
  }
  
  void setEqualizerMode(String mode) {
    print('🎚️ Mengatur equalizer ke mode: $mode');
  }
}

  @override
  void showInfo() {
    print('═══════════════════════════════');
    print('📺 VIDEO TRACK');
    print('═══════════════════════════════');
    print('Judul: $judul');
    print('Resolusi: $resolusiLayar');
    print('Format: $formatVideoTrack');
    displayDuration();
    print('Status: ${isPlaying ? "Playing" : (isPaused ? "Paused" : "Stopped")}');
    print('═══════════════════════════════');
  }
  
  @override
  void displayDuration() {
    int menit = durasi ~/ 60;
    int detik = durasi % 60;
    print('Durasi: ${menit.toString().padLeft(2, '0')}:${detik.toString().padLeft(2, '0')}');
  }
  
  void setQuality(String kualitas) {
    print('📺 Mengubah kualitas video ke: $kualitas');
  }
  
  void seekTo(int detik) {
    if (detik >= 0 && detik <= durasi) {
      currentTime = detik;
      int menit = detik ~/ 60;
      int second = detik % 60;
      print('⏩ Lompat ke ${menit.toString().padLeft(2, '0')}:${second.toString().padLeft(2, '0')}');
    } else {
      print('❌ Waktu tidak valid!');
    }
  }
  
  void fullscreenMode() {
    print('🖥️ Mode layar penuh aktif');
  }
}
