Pengantar Polimorfisme

Dart menyediakan beberapa mekanisme inisialisasi objek untuk berbagai kebutuhan arsitektur:

  1. Arti: "Banyak bentuk" (poly = banyak, morph = bentuk)
  2. Konsep: Objek dari subclass dapat diperlakukan sebagai objek dari superclass
  3. Parameterized Constructor: ClassName(this.param)
  4. Contoh: Semua Kucing , Anjing , Burung adalah Hewan
  5. i.Compile-time polymorphism: Method overloading (tidak didukung Dart)
  6. ii. Runtime polymorphism: Method overriding (didukung Dart)

    print('Menerima panggilan dari $number');
  }

// Kelas Tablet mewarisi Komputer dan menggunakan mixin Kamera & Telepon
class Tablet extends Komputer with Kamera, Telepon {
  bool hasStylus;
  String? cellularNetwork;

  Tablet(String processor, int ram, int storage, this.hasStylus, this.cellularNetwork)
      : super(processor, ram, storage);

  @override
  void makeCall(String number) {
    if (cellularNetwork != null) {
      print('Tablet melakukan panggilan ke $number via $cellularNetwork');
    } else {
      print('Tablet tidak mendukung panggilan suara (WiFi only). Gunakan aplikasi VoIP.');
    }
  }

  @override
  void receiveCall(String number) {
    if (cellularNetwork != null) {
      print('Menerima panggilan dari $number');
    } else {
      print('Tablet tidak bisa menerima panggilan seluler.');
    }
  }

  void useStylus() {
    if (hasStylus) {
      print('Menggunakan stylus untuk menulis di layar.');
    } else {
      print('Tablet ini tidak dilengkapi stylus.');
    }
  }

  void enableCellular() {
    if (cellularNetwork != null) {
      print('Mode seluler diaktifkan. Jaringan: $cellularNetwork');
    } else {
      print('Tablet ini tidak memiliki modem seluler.');
    }
  }

  @override
  void displaySpecs() {
    super.displaySpecs();
    print('Stylus: ${hasStylus ? "Ya" : "Tidak"}');
    print('Jaringan: ${cellularNetwork ?? "WiFi only"}');
  }
}

import 'dart:math';

class MathUtils {
  // 1. Menghitung faktorial
  static int factorial(int n) {
    if (n < 0) {
      throw ArgumentError('Faktorial tidak terdefinisi untuk bilangan negatif');
    }
    if (n == 0 || n == 1) return 1;
    
    int result = 1;
    for (int i = 2; i <= n; i++) {
      result *= i;
    }
    return result;
  }
  
  // 2. Cek bilangan prima
  static bool isPrime(int number) {
    if (number <= 1) return false;
    if (number == 2) return true;
    if (number % 2 == 0) return false;
    
    int sqrtNum = (sqrt(number)).toInt();
    for (int i = 3; i <= sqrtNum; i += 2) {
      if (number % i == 0) return false;
    }
    return true;
  }
  
  // 3. Membulatkan double ke desimal tertentu
  static double roundToDecimal(double value, int decimals) {
    if (decimals < 0) {
      throw ArgumentError('Jumlah desimal tidak boleh negatif');
    }
    double factor = pow(10, decimals).toDouble();
    return (value * factor).round() / factor;
  }
  
  // 4. Menghasilkan bilangan acak integer dalam range [min, max]
  static int randomInt(int min, int max) {
    if (min > max) {
      throw ArgumentError('min harus kurang dari atau sama dengan max');
    }
    final random = Random();
    return min + random.nextInt(max - min + 1);
  }
  
  // 5. Menghasilkan bilangan acak double dalam range [min, max]
  static double randomDouble(double min, double max) {
    if (min > max) {
      throw ArgumentError('min harus kurang dari atau sama dengan max');
    }
    final random = Random();
    return min + random.nextDouble() * (max - min);
  }
  
  // 6. Menghitung rata-rata dari list angka
  static double mean(List numbers) {
    if (numbers.isEmpty) {
      throw ArgumentError('List tidak boleh kosong');
    }
    double sum = 0;
    for (var num in numbers) {
      sum += num.toDouble();
    }
    return sum / numbers.length;
  }
  
  // 7. Menghitung median dari list angka
  static double median(List numbers) {
    if (numbers.isEmpty) {
      throw ArgumentError('List tidak boleh kosong');
    }
    
    List sorted = numbers.map((e) => e.toDouble()).toList();
    sorted.sort();
    int length = sorted.length;
    
    if (length % 2 == 0) {
      // Jika genap, ambil rata-rata dua nilai tengah
      return (sorted[length ~/ 2 - 1] + sorted[length ~/ 2]) / 2;
    } else {
      // Jika ganjil, ambil nilai tengah
      return sorted[length ~/ 2];
    }
  }
}

// ==================== ABSTRACT CLASS ====================
abstract class MediaPlayer {
  String title;
  int duration;
  
  MediaPlayer(this.title, this.duration);
  
  void play();
  void stop();
  void showInfo();
  
  void displayDuration() {
    int menit = duration ~/ 60;
    int detik = duration % 60;
    print('Durasi: ${menit.toString().padLeft(2, '0')}:${detik.toString().padLeft(2, '0')}');
  }
}

// ==================== INTERFACES ====================
abstract interface class Playable {
  void play();
  void resume();
}

abstract interface class Pausable {
  void pause();
  void resumeFromPause();
}

// ==================== CLASS IMPLEMENTASI ====================
class AudioPlayer implements MediaPlayer, Playable, Pausable {
  @override
  String title;
  
  @override
  int duration;
  
  String formatAudio;
  int bitrate;
  bool isPlaying = false;
  bool isPaused = false;
  
  AudioPlayer({
    required this.title,
    required this.duration,
    required this.formatAudio,
    required this.bitrate,
  });
  
  @override
  void play() {
    if (isPaused) {
      resume();
    } else {
      isPlaying = true;
      isPaused = false;
      print('🎵 Memutar audio: $title ($formatAudio, ${bitrate}kbps)');
    }
  }
  
  @override
  void resume() {
    if (isPaused) {
      isPlaying = true;
      isPaused = false;
      print('🎵 Melanjutkan audio: $title');
    }
  }
  
  @override
  void stop() {
    isPlaying = false;
    isPaused = false;
    print('⏹️ Menghentikan audio: $title');
  }
  
  @override
  void pause() {
    if (isPlaying) {
      isPlaying = false;
      isPaused = true;
      print('⏸️ Menjeda audio: $title');
    }
  }
  
  @override
  void resumeFromPause() {
    if (isPaused) {
      isPlaying = true;
      isPaused = false;
      print('▶️ Melanjutkan audio dari jeda: $title');
    }
  }
  
  @override
  void showInfo() {
    print('═══════════════════════════════');
    print('📻 AUDIO PLAYER');
    print('═══════════════════════════════');
    print('Judul: $title');
    print('Format: $formatAudio');
    print('Bitrate: ${bitrate}kbps');
    displayDuration();
    print('Status: ${isPlaying ? "Playing" : (isPaused ? "Paused" : "Stopped")}');
    print('═══════════════════════════════');
  }
  
  @override
  void displayDuration() {
    int menit = duration ~/ 60;
    int detik = duration % 60;
    print('Durasi: ${menit.toString().padLeft(2, '0')}:${detik.toString().padLeft(2, '0')}');
  }
  
  void setVolume(int volume) {
    print('🔊 Mengatur volume ke $volume%');
  }
  
  void setEqualizer(String mode) {
    print('🎚️ Mengatur equalizer ke mode: $mode');
  }
}

    }

  @override

    int menit = duration ~/ 60;
    int detik = duration % 60;
    print('Durasi: ${menit.toString().padLeft(2, '0')}:${detik.toString().padLeft(2, '0')}');
